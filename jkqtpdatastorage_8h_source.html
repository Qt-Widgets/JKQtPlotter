71 | 0 | 1 | 2 | + N-1 || 0 | 1 | 2 | + N-1 |

432  /** \brief add one external column to the datastore. It will be filled with the contents of vector \a data.
433  *
434  * \tparam TContainer datatype of the container, which need to support standard C++ iterators and the function \c size(). The contents needs to be convertible to double.
435  * \param data data vector to be copied
436  * \param name name for the column
437  * \return the ID of the newly created column
438  *
439  * \code
440  * QVector<double> X, Y;
441  * const int Ndata=100;
442  * for (int i=0; i<Ndata; i++) {
443  * const double x=double(i)/double(Ndata)*8.0*M_PI;
444  * X<<x;
445  * Y<<sin(x);
446  * }
447  * plot.addGraph(linegraph=new JKQTPXYLineGraph(&plot));
448  * // by calling JKQTPDatastore::addCopiedColumn() the data is COPIED from the vector into the datastore
449  * linegraph->setXColumn(datastore->addCopiedColumn(X, "x"));
450  * linegraph->setYColumn(datastore->addCopiedColumn(Y, "y"));
451  * \endcode
452  *
453  * \see \ref JKQTPlotterBasicJKQTPDatastore
454  */
455  template <typename TContainer>
456  size_t addCopiedColumn(const TContainer& data, const QString& name=QString("")) {
457  const size_t N=static_cast<size_t>(data.size());
458  double* d=static_cast<double*>(malloc(static_cast<size_t>(N)*sizeof(double)));
459  if (N>0) {
460  size_t r=0;
461  for (auto it=data.begin(); it!=data.end(); ++it) {
462  d[r]=jkqtp_todouble(*it);
463  r++;
464  }
465  }
466  size_t itemid=addInternalItem(d, N);
467  return addColumnForItem(itemid, 0, name);
468  }
